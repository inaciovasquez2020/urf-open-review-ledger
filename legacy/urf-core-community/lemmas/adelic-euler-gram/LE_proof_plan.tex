\section*{Goal}
For every elliptic curve $E/\mathbb{Q}$, prove the strict spectral gap
\[
\lambda_1(E):=\inf_{\phi\perp\ker\mathcal L_E}
\frac{\langle \mathcal L_E\phi,\phi\rangle}{\|\phi\|^2} > 0.
\]

\section*{Equivalent coercivity form}
There exists $c(E)>0$ such that for all $P\in E(\mathbb{Q})\setminus E(\mathbb{Q})_{\mathrm{tors}}$,
\[
\hat h(P)\ge c(E)\,\|P\|^2.
\]

\section*{Required missing lemma}
Construct a norm $\|\cdot\|$ on $E(\mathbb{Q})\otimes_{\mathbb{Z}}\mathbb{R}$ and prove:
\[
\inf_{0\neq v\in (E(\mathbb{Q})\otimes\mathbb{R})}
\frac{\langle v,v\rangle_{\mathrm{NT}}}{\|v\|^2} > 0,
\]
where $\langle\cdot,\cdot\rangle_{\mathrm{NT}}$ is the Néron--Tate bilinear form.
