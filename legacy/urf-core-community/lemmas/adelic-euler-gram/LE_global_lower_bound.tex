\begin{openproblem}[Global Height Lower Bound]
Prove that there exists a constant $c(E)>0$ such that for all
$P \in E(\mathbb{Q}) \setminus E(\mathbb{Q})_{\mathrm{tors}}$,
\[
\hat h(P) \ge c(E).
\]
Equivalently, the set of Néron--Tate heights of non-torsion rational points
has a positive minimum.
\end{openproblem}
