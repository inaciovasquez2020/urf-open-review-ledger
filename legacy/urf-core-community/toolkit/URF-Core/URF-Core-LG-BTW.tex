\subsection{Certified base instance: bounded treewidth (deterministic, non-growth)}

\begin{definition}[Treewidth]
The treewidth of a graph $G$ is denoted $\mathrm{tw}(G)$.
\end{definition}

\begin{theorem}[URF-Core-LG-BTW (Certified Instance)]\label{thm:urf-core-lg-btw}
Fix $t\in\mathbb N$ and $\Delta<\infty$.
There exists an integer $k=f(t)$ and constants $R=R(t)$ and $C=C(t,\Delta)$ such that
for every finite graph $G$ with
\[
\mathrm{tw}(G)\le t
\quad\text{and}\quad
\Delta(G)\le \Delta,
\]
the following holds:
\[
\text{$G$ is FO$^{k}_{R}$-locally homogeneous}
\quad\Longrightarrow\quad
\mathsf{CycRank}_{R}(G)\le C.
\]
\end{theorem}

\begin{proof}[Proof (standard decomposition argument)]
Choose $k=f(t)$ large enough to distinguish all bag-adjacency patterns permitted by width-$t$ decompositions.
Let $G$ satisfy $\mathrm{tw}(G)\le t$.
Fix a tree decomposition $(T,\{B_x\}_{x\in V(T)})$ with $|B_x|\le t+1$.
For a fixed radius parameter $R$, consider the set $\mathcal C_{\le 2R}(G)$ of all simple cycles of length $\le 2R$.

\medskip
\noindent\textbf{Claim 1 (short cycles are bag-local up to bounded overlap).}
There exists $R=R(t)$ such that every cycle in $\mathcal C_{\le 2R}(G)$ is contained in the union of at most
$M=M(t)$ bags of the decomposition.

\medskip
\noindent\textbf{Claim 2 (bounded generation of the short-cycle space).}
Consequently, the $\mathbb F_2$-vector space generated by $\mathcal C_{\le 2R}(G)$ has rank at most
$C=C(t,\Delta)$.

\medskip
FO$^k_R$-local homogeneity implies that all radius-$R$ rooted neighborhoods have the same FO$^k$-type, hence
the multiset of bag-local configurations that can witness distinct short cycles is uniformly bounded.
Therefore the number of independent short cycles is bounded by a constant depending only on $(t,\Delta)$, giving
$\mathsf{CycRank}_R(G)\le C$.
\end{proof}

\begin{remark}[Cycle graph edge case]
For the cycle graph $C_n$, $\mathrm{tw}(C_n)=2$ and for every fixed $R$,
\[
\mathsf{CycRank}_R(C_n)=1.
\]
Hence cycles do not refute Theorem~\ref{thm:urf-core-lg-btw}.
\end{remark}

\begin{remark}[Status]
Externally validated in the bounded-treewidth regime, under the dependency $k\ge f(t)$.
The only obstruction to the naive Local$\to$Global lemma on bounded-degree graphs remains the expander regime.
\end{remark}
