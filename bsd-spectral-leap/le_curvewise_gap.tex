\section*{Curve-wise Spectral Gap for the Néron--Tate Form}

\begin{definition}[Height pairing kernel]
For an elliptic curve $E/\mathbb Q$, define
\[
K_E(P,Q) := \hat h(P+Q) - \hat h(P) - \hat h(Q).
\]
\end{definition}

\begin{lemma}[Positive definiteness]
The Néron--Tate height induces a positive definite quadratic form on
$E(\mathbb Q)/E(\mathbb Q)_{\mathrm{tors}}$.
\end{lemma}

\begin{proof}
For fixed $E/\mathbb Q$, the Néron--Tate height $\hat h$ is a quadratic form on
$E(\mathbb Q)\otimes_{\mathbb Z}\mathbb R$ whose kernel is exactly
$E(\mathbb Q)_{\mathrm{tors}}\otimes_{\mathbb Z}\mathbb R$.
Hence it descends to a positive definite quadratic form on
$E(\mathbb Q)/E(\mathbb Q)_{\mathrm{tors}}$.
\end{proof}

\begin{lemma}[Lattice minimum]
A positive definite quadratic form on a lattice attains a strictly positive minimum on nonzero lattice points.
\end{lemma}

\begin{proof}
Let $Q$ be a positive definite quadratic form on a lattice $\Lambda \cong \mathbb Z^r$.
The set $\{x\in\Lambda : Q(x)\le 1\}$ is finite.
Hence $Q$ attains a strictly positive minimum on $\Lambda\setminus\{0\}$.
\end{proof}

\begin{theorem}[Curve-wise spectral gap]
For fixed $E/\mathbb Q$,
\[
\inf_{P \in E(\mathbb Q)\setminus E(\mathbb Q)_{\mathrm{tors}}} \hat h(P) > 0 .
\]
\end{theorem}

\begin{proof}
By the lattice minimum lemma, the Néron--Tate form $\hat h$ has a strictly positive
minimum $\lambda(E)>0$ on
$E(\mathbb Q)\setminus E(\mathbb Q)_{\mathrm{tors}}$.
Hence for any nonzero rational point $P$,
\[
\hat h(P)\ge \lambda(E).
\]
This yields a uniform spectral gap on each fixed curve $E/\mathbb Q$.
\end{proof}

\begin{proof}
By positive definiteness of the Néron--Tate height and the lattice minimum lemma,
there exists $\lambda(E)>0$ such that
$\hat h(P)\ge \lambda(E)$ for all
$P\in E(\mathbb Q)\setminus E(\mathbb Q)_{\mathrm{tors}}$.
The curve-wise spectral gap lemma identifies $\lambda(E)$ as a uniform lower bound
for the quadratic form on the non-torsion lattice.
\end{proof}
