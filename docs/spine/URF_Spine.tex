\section*{Gödel-Style Numbered Laws}

\textbf{Axiom 1 (URF-0.0 Finite Realizability).}  
All information-bearing objects admit a finite realization.

\textbf{Axiom 2 (URF-0.1 Locality).}  
There exists $r$ such that
\[
Y_t = f_t(N_r(X), Y_{<t})
\]
for all $t$.

\textbf{Axiom 3 (URF-0.2 Capacity).}  
There exists $C<\infty$ such that for all $T$,
\[
I(X;Y_{1:T}) \le C.
\]

\textbf{Axiom 4 (URF-0.3 Entropy Non-Amplification).}  
For all $t$,
\[
I(X;Y_t \mid Y_{<t}) \le K
\]
for a universal constant $K$.

\textbf{Axiom 5 (URF-0.4 No Global Reconstruction).}  
No admissible refinement reconstructs unbounded global structure from local views.

\bigskip

\textbf{Axiom 6 (URF-0.5 Exchange Correlation Capacity Bound).}  
For any systems $A,B$ interacting only via a mediator $M$, if $M$ is
finite-dimensional or quasi-local with exponential clustering, then
there exists a constant $\mathrm{Cap}(M)<\infty$ such that
\[
\forall t:\quad I(A:B)_t \le \mathrm{Cap}(M).
\]

\bigskip

\textbf{Axiom 7 (URF-0.6 Support Rigidity).}  
For all $r$ there exists $R$ such that
\[
\mathrm{FO}^k_r(x) \equiv \mathrm{FO}^k_r(y)
\;\Rightarrow\;
\mathrm{FO}^k_R(N_R(x)) \equiv \mathrm{FO}^k_R(N_R(y)).
\]

\textbf{Axiom 8 (URF-0.7 Irreversibility).}  
Information erasure produces entropy at least $k_B \ln 2$ per bit.

\textbf{Axiom 9 (URF-0.8 Non-Viotropy).}  
No admissible process satisfies
\[
\lim_{E\to 0} \frac{I}{E} = \infty.
\]

\textbf{Axiom 10 (URF-0.9 Rigidity–Collapse).}  
Persistent local indistinguishability implies global structural collapse.

\bigskip

\textbf{Axiom 11 (URF-1.0 No Free Genesis).}  
There exists a universal constant $K_G$ such that for all $t$,
\[
H_{\mathrm{ext}}(Y_{1:t}) \le H_{\mathrm{ext}}(Y_{1:t-1}) + K_G.
\]

\textbf{Axiom 12 (URF-1.1 Extractor Completeness).}  
Every admissible global invariant is FO$^k$-definable from the transcript:
\[
\forall G\in\mathcal G_{\mathrm{adm}}\;\exists f\in\mathcal F_{\mathrm{adm}}\;\forall t:\;
G(\mathrm{history}_{\le t}) = f(Y_{1:t}).
\]

\section*{Genesis and Extractable Entropy}

\begin{definition}[Admissible Extractors]
Fix constants $k,r$. Define
\[
\mathcal F_{\mathrm{adm}} := 
\{ f \mid f(Y_{1:t}) = \mathrm{Code}(\mathrm{Th}^{k,r}(\mathcal I(Y_{1:t}))) \}
\]
for some FO$^k$ interpretation $\mathcal I$.
\end{definition}

\begin{definition}[Extractable Entropy]
\[
H_{\mathrm{ext}}(t) := \sup_{f\in\mathcal F_{\mathrm{adm}}}
H_{\mathrm{Shannon}}(f(Y_{1:t})).
\]
\end{definition}

\begin{lemma}[VCDI Dominates Extractable Entropy]
Assume URF-0.2. Then for all $t$,
\[
H_{\mathrm{ext}}(t) \le C.
\]
\end{lemma}

\begin{proof}
For any $f\in\mathcal F_{\mathrm{adm}}$,
\[
H(f(Y_{1:t})) \le I(X;f(Y_{1:t}))
\le I(X;Y_{1:t}) \le C,
\]
by data processing and URF-0.2. Taking supremum yields the result.
\end{proof}

\begin{theorem}[Genesis Wall]
Assume URF-0.2, URF-1.0, and URF-1.1. Then
\[
\sup_t H_{\mathrm{ext}}(t) < \infty.
\]
\end{theorem}
