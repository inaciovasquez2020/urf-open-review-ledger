\section*{EF--Game Constants for CLR}

\begin{lemma}[Bounded configuration space]
For fixed $k,\Delta,R$, the number of $k$-pebble EF configurations on radius-$R$ neighborhoods of $\Delta$-bounded graphs is finite.
\end{lemma}

\begin{lemma}[Uniform round bound]
There exists a round bound $r=r(k,\Delta,R)$ such that FO$^k$ equivalence on radius-$R$ balls is witnessed within $r$ rounds.
\end{lemma}

\begin{proposition}[Explicit constants interface]
All constants used in CLR depend only on $(k,\Delta,R)$ through the configuration bound and the round bound above.
\end{proposition}
