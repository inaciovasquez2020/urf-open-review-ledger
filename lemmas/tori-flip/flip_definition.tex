\begin{definition}[Tori Flip]
Let $X$ be a smooth projective variety and $J^k(X)$ its intermediate Jacobian.
For a real subtorus $T \subset J^k(X)$ define
\[
\mathsf{Flip}(T)
:= \overline{\langle T \cap J^k(X)(\mathbb{Q}) \rangle}_{\mathbb{R}},
\]
the smallest rational Hodge substructure containing $T$.
\end{definition}
