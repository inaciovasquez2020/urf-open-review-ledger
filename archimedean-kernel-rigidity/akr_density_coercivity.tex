\section*{Density--Coercivity of the Archimedean Kernel}

\begin{definition}[Paley--Wiener core]
Let $D$ be the Paley--Wiener subspace of $L^2(\mathbb R)$ consisting of test functions with compactly supported Fourier transform.
\end{definition}

\begin{lemma}[Diagonal dominance on the core]
For $f\in D$ with frequency support bounded away from zero, the diagonal term of the explicit formula satisfies
\[
\langle \mathcal E_{\mathrm{diag}} f,f\rangle \ge c_0 \|f\|_2^2
\]
for some $c_0>0$.
\end{lemma}

\begin{proof}
On the Paley--Wiener core $D$, the explicit formula decomposes the quadratic form
$\langle \mathcal E f,f\rangle$ into a diagonal prime-sum term and oscillatory terms.
For frequency support bounded away from zero, the diagonal contribution is a finite,
absolutely convergent sum with strictly positive coefficients.
Uniform lower bounds follow from positivity of the prime weights and Plancherel,
yielding $\langle \mathcal E_{\mathrm{diag}} f,f\rangle \ge c_0 \|f\|_2^2$.
\end{proof}

\begin{lemma}[Oscillatory control on the core]
On $D$, the oscillatory term satisfies
\[
|\langle \mathcal E_{\mathrm{osc}} f,f\rangle| \le \varepsilon(R)\|f\|_2^2
\]
with $\varepsilon(R)\to 0$ as the frequency cutoff $R\to\infty$.
\end{lemma}

\begin{proof}
On the Paley--Wiener core $D$, the oscillatory term arises from zero interactions in the
explicit formula. For compactly supported Fourier transform and frequency cutoff $R$,
standard explicit-formula estimates give uniform bounds on each oscillatory component.
Summing these bounds and applying Plancherel yields
\[
|\langle \mathcal E_{\mathrm{osc}} f,f\rangle| \le \varepsilon(R)\|f\|_2^2
\]
with $\varepsilon(R)\to 0$ as $R\to\infty$.
\end{proof}

\begin{theorem}[Density--coercivity]
For $R$ sufficiently large and $f\in D$ orthogonal to $\ker(\mathcal E)$,
\[
\langle \mathcal E f,f\rangle \ge c\|f\|_2^2 .
\]
\end{theorem}
